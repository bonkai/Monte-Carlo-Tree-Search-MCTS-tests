```python
import logging
import json
import requests
from typing import Dict, Any
import yaml
import os
import time

# Setup Logging Configuration
logging.basicConfig(level=logging.DEBUG,
                    format='%(asctime)s %(levelname)s: %(message)s',
                    datefmt='%Y-%m-%d %H:%M:%S')


class ConfigManager:
    def __init__(self, config_file="config.yaml"):
        logging.info("Loading configuration from file...")
        self.config = None
        
        if os.path.exists(config_file):
            with open(config_file) as file:
                self.config = yaml.safe_load(file)
        else:
            self.config = {
                'api_url': os.getenv('API_URL', ''),
                'api_key': os.getenv('API_KEY', '')
            }
        
        logging.info(f"Configuration loaded successfully: {self.config}")

    def get_api_url(self):
        return self.config.get('api_url', '')

    def get_api_key(self):
        return self.config.get('api_key', '')


class RequestHandler:
    def __init__(self, config_manager: ConfigManager):
        logging.info("Initializing request handler...")
        self.config_manager = config_manager
        self.api_url = self.config_manager.get_api_url()
        self.api_key = self.config_manager.get_api_key()

    def send_request(self, prompt):
        headers = {'Content-Type': 'application/json'}
        if self.api_key:
            headers['Authorization'] = f'Bearer {self.api_key}'

        logging.debug(f"Sending request to API: URL={self.api_url}, Headers={headers}")
        
        response = requests.post(
            url=self.api_url,
            data=json.dumps({'prompt': prompt}),
            headers=headers
        )
        
        return self.validate_response(response)

    def validate_response(self, response):
        if not response.ok:
            logging.error(f"Failed to get response from API: {response.status_code}")
            raise Exception("API Request Failed")

        try:
            data = response.json()
            logging.debug(f"Received JSON Response: {data}")
        except ValueError as e:
            logging.error(f"Malformed JSON Response: {e}")
            raise

        return data


class UserInterface:
    def __init__(self, request_handler: RequestHandler):
        self.request_handler = request_handler
        self.start_time = time.time()  # Record the start time here

    def run(self):
        logging.info("Starting application...")
        while True:
            try:
                question = self.get_user_input()
                if not question.strip():
                    continue

                db_prompt = question  # Store user input in db_prompt
                response_data = self.request_handler.send_request(db_prompt)
                print(f"Response: {response_data}")
            except Exception as e:
                logging.error(f"Error occurred: {e}")

    def get_user_input(self):
        return input("Enter your question or command (type 'exit' to quit): ")

    def end_program(self):
        duration = time.time() - self.start_time
        logging.info(f"Program execution time: {duration:.2f} seconds")
        print(f"\nProgram executed in {duration:.2f} seconds.")


if __name__ == "__main__":
    config_manager = ConfigManager()
    request_handler = RequestHandler(config_manager)
    ui = UserInterface(request_handler)
    
    try:
        ui.run()
    finally:
        ui.end_program()
```

### Explanation

1. **User Input Handling**: The `get_user_input` method captures the user's input and returns it.
2. **Storing Prompt in `db_prompt`**: Inside the `run` method, we store the user's input (`question`) in a variable named `db_prompt`.
3. **Send Request with `db_prompt`**: We use `db_prompt` as the prompt for sending the request to the API.

This ensures that the user's input is stored and used correctly throughout your application. The elapsed time calculation remains intact, allowing you to measure how long the program runs.
