// script.js
document.addEventListener("DOMContentLoaded", function() {
    const nextButtons = document.querySelectorAll('.next');

    nextButtons.forEach(button => {
        button.addEventListener('click', (event) => {
            event.preventDefault();
            // Navigate to the next page
            window.location.href = `page${parseInt(window.location.pathname.split('/').pop().replace('.html', '')) + 1}.html`;
        });
    });

    // Handle the sign-up page navigation
    if (window.location.pathname.includes('sign-up.html')) {
        document.querySelector('.next').addEventListener('click', function() {
            event.preventDefault();
            alert("Thank you for signing up! We will contact you soon.");
        });
    }
});
```

function getCurrentPage() {
    const urlPath = window.location.pathname.split('/').pop();
    return parseInt(urlPath.replace('.html', ''), 10);

function navigateToNextPage(currentPage) {
    let nextPageUrl;

    if (currentPage === 4) { // Last page, redirect to sign-up
        nextPageUrl = 'sign-up.html';
    } else {
        nextPageUrl = `page${currentPage + 1}.html`;

    window.location.href = nextPageUrl;
}

document.addEventListener('DOMContentLoaded', function() {

document.addEventListener('DOMContentLoaded', function() {
    const buttons = document.querySelectorAll('.nextButton');

    buttons.forEach(button => {
        button.addEventListener('click', () => {
            let currentPage = parseInt(window.location.pathname.split('/').pop().replace('.html', ''));

            if (currentPage === 1) {
                window.location.href = 'page2.html';
            } else if (currentPage === 2) {
                window.location.href = 'page3.html';
            } else if (currentPage === 3) {
                window.location.href = 'page4.html';
            } else if (currentPage === 4) {
                window.location.href = 'sign-up.html';
            }
        });
    });

    // Handling direct navigation to specific pages
    const urlParams = new URLSearchParams(window.location.search);
    let directNavigationPage = urlParams.get('direct');

    if (directNavigationPage !== null && parseInt(directNavigationPage) === 4) {
        window.location.href = 'sign-up.html';
    }
});
```
