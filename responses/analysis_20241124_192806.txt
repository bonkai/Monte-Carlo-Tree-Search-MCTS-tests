```html
<script>
const checkoutSteps = [
  { id: 'cart', label: 'Cart' },
  { id: 'shipping-address', label: 'Shipping Address' },
  { id: 'billing-information', label: 'Billing Information' },
  { id: 'payment-method', label: 'Payment Method' },
  { id: 'review-order', label: 'Review Order' }

let currentStepIndex = 0;

function updateProgressBar(stepIndex) {
  const progressBar = document.querySelector('.progress-bar');
  // Update progress bar width and step labels
}

function showStepIndicators() {
  const indicatorsList = document.getElementById('step-indicators');
  indicatorsList.innerHTML = ''; // Clear previous steps
  checkoutSteps.forEach((step, index) => {
    const li = document.createElement('li');
    li.className = `step ${index === currentStepIndex ? 'active' : ''}`;
    li.innerText = step.label;
    indicatorsList.appendChild(li);
  });
}

function moveToNextStep() {
  if (currentStepIndex < checkoutSteps.length - 1) {
    currentStepIndex++;
    updateProgressBar(currentStepIndex);
    showFormForStep(checkoutSteps[currentStepIndex].id);
    showStepIndicators();
  }
}
</script>
```
