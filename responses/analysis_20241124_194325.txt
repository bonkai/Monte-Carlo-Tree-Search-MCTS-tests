import requests
import json
import logging

# Setup logging configuration
logging.basicConfig(level=logging.DEBUG, format='%(asctime)s %(levelname)s: %(message)s', datefmt='%Y-%m-%d %H:%M:%S')

def get_api_response(prompt):
    API_URL = "https://api.ollama.com/v1/ask"  # Replace with the actual Ollama API endpoint
    API_KEY = "your-api-key-here"  # Replace with your actual API key

    headers = {
        'Content-Type': 'application/json',
        'Authorization': f'Bearer {API_KEY}',
    }

    data = {
        'prompt': prompt,
    }

    try:
        logging.info(f"Sending request to Ollama API: {data}")
        response = requests.post(API_URL, headers=headers, json=data)
        response.raise_for_status()
        result = response.json()

        if 'answer' in result:
            answer = result['answer']
            logging.info(f"Received response from Ollama API: {answer}")
            return answer
        else:
            logging.error("Response missing expected key 'answer'")
            raise Exception("Unexpected response format")

    except requests.exceptions.RequestException as e:
        logging.exception(f"Error occurred during request to Ollama API: {e}")
        return None

def main():
    prompt = input("Enter your question or prompt: ")
    answer = get_api_response(prompt)
    
    if answer is not None:
        print(f"Response from the model: {answer}")

if __name__ == "__main__":
    main()
```

import logging
import json
from typing import Dict, Any
import requests
import yaml
import os

# Setup Logging Configuration
logging.basicConfig(level=logging.DEBUG,
                    format='%(asctime)s %(levelname)s: %(message)s',
                    datefmt='%Y-%m-%d %H:%M:%S')

class ConfigManager:
    def __init__(self, config_file="config.yaml"):
        self.config = self.load_config(config_file)
    
    @staticmethod
    def load_config(file_name):
        if os.path.exists(file_name):
            with open(file_name) as f:
                return yaml.safe_load(f)
        else:
            raise FileNotFoundError("Configuration file not found!")

class RequestHandler:
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.base_url = config['api']['url']
    
    def send_request(self, prompt: str) -> dict:
        url = f"{self.base_url}/query"
        headers = {
            'Content-Type': 'application/json',
            'Authorization': f"Bearer {self.config['api']['key']}"
        }
        
        response = requests.post(url, json={"prompt": prompt}, headers=headers)
        if response.status_code == 200:
            return response.json()
        else:
            logging.error(f"Request failed with status code: {response.status_code}")
            raise Exception("API request failed")

class UserInterface:
    def __init__(self):
        self.config_manager = ConfigManager()
        self.request_handler = RequestHandler(self.config_manager.config)
    
    def run(self) -> None:
        while True:
            prompt = input("Enter your question (or 'exit' to quit): ")
            if prompt.lower() == "exit":
                break
            try:
                response_data = self.request_handler.send_request(prompt)
                print(f"Response: {response_data['answer']}")
            except Exception as e:
                logging.error(str(e))
                print("An error occurred. Please check the logs for more details.")

if __name__ == "__main__":
    ui = UserInterface()
    ui.run()
```

```python
import logging
import json
import requests
from typing import Dict, Any
import yaml
import os


# Setup Logging Configuration
logging.basicConfig(level=logging.DEBUG,
                    format='%(asctime)s %(levelname)s: %(message)s',
                    datefmt='%Y-%m-%d %H:%M:%S')


class ConfigManager:
    def __init__(self, config_file="config.yaml"):
        with open(config_file) as file:
            self.config = yaml.safe_load(file)

    def get_api_url(self):
        return self.config.get('api_url', '')

    def get_api_key(self):
        return self.config.get('api_key', '')


class RequestHandler:
    def __init__(self, config_manager: ConfigManager):
        self.config_manager = config_manager
        self.api_url = self.config_manager.get_api_url()
        self.api_key = self.config_manager.get_api_key()

    def send_request(self, prompt):
        headers = {'Content-Type': 'application/json'}
        if self.api_key:
            headers['Authorization'] = f'Bearer {self.api_key}'

        response = requests.post(
            url=self.api_url,
            data=json.dumps({'prompt': prompt}),
            headers=headers
        )
        
        return self.validate_response(response)

    def validate_response(self, response):
        if not response.ok:
            logging.error(f"Failed to get response from API: {response.status_code}")
            raise Exception("API Request Failed")

        try:
            data = response.json()
        except ValueError as e:
            logging.error(f"Malformed JSON Response: {e}")
            raise

        return data


class UserInterface:
    def __init__(self, request_handler: RequestHandler):
        self.request_handler = request_handler

    def run(self):
        while True:
            try:
                user_input = self.get_user_input()
                if not user_input.strip():
                    continue

                if user_input.lower() == 'exit':
                    logging.info("Exiting the program.")
                    break

                response_data = self.request_handler.send_request(user_input)
                print(f"Response: {response_data}")
            except Exception as e:
                logging.error(f"Error occurred: {e}")

    def get_user_input(self):
        return input("Enter your question or command (type 'exit' to quit): ")


if __name__ == "__main__":
    config_manager = ConfigManager()
    request_handler = RequestHandler(config_manager)
    ui = UserInterface(request_handler)
    ui.run()
```
