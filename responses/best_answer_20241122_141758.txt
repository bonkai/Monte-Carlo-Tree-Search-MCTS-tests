### JavaScript (scripts.js)

```javascript
document.addEventListener("DOMContentLoaded", function() {
    const steps = document.querySelectorAll(".step-step");
    let currentStep = 0;

    function showCurrentStep(index) {
        steps.forEach((step, i) => {
            step.style.display = i === index ? "block" : "none";
        });
    }

    function nextStep() {
        if (currentStep < steps.length - 1) {
            currentStep++;
            showCurrentStep(currentStep);
        }
    }

    function prevStep() {
        if (currentStep > 0) {
            currentStep--;
            showCurrentStep(currentStep);
        }
    }

    document.getElementById("next-step-1").addEventListener("click", () => nextStep());
    document.getElementById("prev-step-2").addEventListener("click", () => prevStep());

    // Stripe Elements setup
    const stripe = Stripe('YOUR_STRIPE_PUBLIC_KEY');
    const elements = stripe.elements();
    
    const cardElement = elements.create('card', { style: { base: { fontSize: '16px' } } });
    cardElement.mount('#payment-form');

    document.getElementById("place-order").addEventListener("click", () => {
        stripe.handleCardPayment(
            "client_secret",
            cardElement
        ).then(function(result) {
            if (result.error) {
                alert(result.error.message);
            } else {
                // Order placed successfully, show confirmation
                console.log('Order placed');
            }
        });
    });

    // Show initial step
    showCurrentStep(currentStep);

});
```

   - The logic ensures that users can navigate through each step of the checkout process sequentially.
